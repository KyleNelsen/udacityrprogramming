colnames(df_starts) <- c("Station", "Count_starts")
colnames(df_ends) <- c("Station", "Count_ends")

# Merge data frames
combined_df <- merge(df_starts, df_ends, by = "Station", all = TRUE)

# Replace NA values with 0
